import numpy as np
import tensorflow as tf
from tensorflow import keras
from tensorflow.keras.datasets import reuters
from tensorflow.keras import models, layers
from tensorflow.keras.utils import to_categorical
import matplotlib.pyplot as plt

# Load the dataset with top 10,000 most frequent words
(train_data, train_labels), (test_data, test_labels) = reuters.load_data(num_words=10000)

print("Training entries:", len(train_data))
print("Test entries:", len(test_data))
print("Example data:", train_data[0])
print("Example label:", train_labels[0])

# Function to convert list of integers to one-hot encoded vector
def vectorize_sequences(sequences, dimension=10000):
    results = np.zeros((len(sequences), dimension))
    for i, sequence in enumerate(sequences):
        results[i, sequence] = 1.0
    return results

# Vectorize input data
x_train = vectorize_sequences(train_data)
x_test = vectorize_sequences(test_data)

# One-hot encode labels (46 classes)
y_train = to_categorical(train_labels)
y_test = to_categorical(test_labels)

# Model Architecture
model = models.Sequential()
model.add(layers.Dense(64, activation='relu', input_shape=(10000,)))
model.add(layers.Dense(64, activation='relu'))
model.add(layers.Dense(46, activation='softmax'))

# Compile
model.compile(
    optimizer='rmsprop',
    loss='categorical_crossentropy',
    metrics=['accuracy']
)

# Validation Split
x_val = x_train[:1000]
partial_x_train = x_train[1000:]
y_val = y_train[:1000]
partial_y_train = y_train[1000:]

# Train
history = model.fit(
    partial_x_train,
    partial_y_train,
    epochs=20,
    batch_size=512,
    validation_data=(x_val, y_val)
)

# Extract Metrics
loss = history.history['loss']
val_loss = history.history['val_loss']
acc = history.history['accuracy']
val_acc = history.history['val_accuracy']
epochs = range(1, len(loss) + 1)

# Plot loss
plt.plot(epochs, loss, 'bo', label='Training loss')
plt.plot(epochs, val_loss, 'b', label='Validation loss')
plt.title('Training and validation loss')
plt.xlabel('Epochs')
plt.ylabel('Loss')
plt.legend()
plt.show()

# Plot accuracy
plt.plot(epochs, acc, 'bo', label='Training acc')
plt.plot(epochs, val_acc, 'b', label='Validation acc')
plt.title('Training and validation accuracy')
plt.xlabel('Epochs')
plt.ylabel('Accuracy')
plt.legend()
plt.show()

# Evaluate model
results = model.evaluate(x_test, y_test)
print("Test Loss, Test Accuracy:", results)

# Step 1: Get predictions
predictions = model.predict(x_test)

# Step 2: Convert to class labels
predicted_labels = np.argmax(predictions, axis=1)

# Convert actual labels
actual_labels = np.argmax(y_test, axis=1)

# Print comparisons for first 10 samples
for i in range(10):
    print(f"Sample {i+1}:")
    print(f"  Actual Label     : {actual_labels[i]}")
    print(f"  Predicted Label  : {predicted_labels[i]}")
    print(f"  Match?           : {'Yes' if actual_labels[i] == predicted_labels[i] else 'No'}")
    print()

# t-SNE Visualization
from sklearn.manifold import TSNE

# Predict features (output of final layer â†’ (2246, 46))
features = model.predict(x_test)

# Reduce to 2D
tsne = TSNE(n_components=2, random_state=42)
tsne_results = tsne.fit_transform(features)

# Convert one-hot to labels
if y_test.ndim > 1:
    y_test_labels = np.argmax(y_test, axis=1)
else:
    y_test_labels = y_test

# Plot t-SNE
plt.figure(figsize=(10, 8))
scatter = plt.scatter(
    tsne_results[:, 0], tsne_results[:, 1],
    c=y_test_labels, cmap='tab20', alpha=0.7
)
plt.colorbar(scatter, label='Class Label')
plt.title('t-SNE Visualization of Reuters News Articles')
plt.grid(True)
plt.show()
